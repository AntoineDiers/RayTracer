71 public OpenCvSharp.Mat Run(in Scene scene)
72 {
73 int width = scene.camera.GetResolution().Width;
74 int height = scene.camera.GetResolution().Height;
75 int nPixels = width * height;
76
77 RayTracerLib.Timer timer = new(4);
78 timer.Restart("Loading scene on the GPU ...");
79
80 // Load the scene on the GPU
81 SceneLoader.LoadScene(scene, accelerator,
82 out var raysBuffer,
83 out var trianglesBuffer,
84 out var subMeshesBuffer,
85 out var lightSourcesBuffer,
86 out var materialsBuffer,
87 out var texturesDataBuffer,
88 out var texturesMetadataBuffer,
89 out var pixelsBuffer);
90
91 timer.Print("");
92 timer.Restart("Running Photon mapping ...");
93
94 // Run Photon Mapping
95 List<Photon> photons = PhotonMapper.Run(scene,
96 accelerator,
97 trianglesBuffer,
98 subMeshesBuffer,
99 materialsBuffer);
100
101 timer.Print("");
102 timer.Restart("Constructing Photons tree ...");
103
104 // Construct the photons tree and load it on the GPU
105 PhotonTree photonsTree = new(photons);
106 PhotonsLoader.LoadPhotonsTree(photonsTree, accelerator, nPixels,
107 out MemoryBuffer1D<GpuPhotonTreeNode, Stride1D.Dense> photonsTreeBuffer,
108 out MemoryBuffer1D<GpuPhotonNeighbor, Stride1D.Dense> photonsNearestNeighborsBuffer,
109 out MemoryBuffer1D<byte, Stride1D.Dense> photonsSearchStateBuffer);
110
111 timer.Print("");
112 timer.Restart("Running main raytracing pass ...");
113
114 // Run the main kernel
115 kernel(nPixels,
116 raysBuffer.View,
117 trianglesBuffer.View,
118 subMeshesBuffer.View,
119 lightSourcesBuffer.View,
120 materialsBuffer.View,
121 texturesDataBuffer.View,
122 texturesMetadataBuffer.View,
123 photonsTreeBuffer.View,
124 photonsNearestNeighborsBuffer.View,
125 photonsSearchStateBuffer.View,
126 pixelsBuffer.View);
127
128 // Retrieve the data
129 Vector3D[] bufferOut = new Vector3D[pixelsBuffer.Length];
130 pixelsBuffer.CopyToCPU(bufferOut);
131
132 timer.Print("");
133 timer.Restart("Generating the output image ...");
134
135 // Generate the image
136 OpenCvSharp.Mat res = new(scene.camera.GetResolution(), MatType.CV_8UC3, new(255, 255, 255));
137 for (int i = 0; i < bufferOut.Length; i++)
138 {
139 int col = i % width;
140 int row = i / width;
141 res.At<Vec3b>(row, col) = ConvertColor(bufferOut[i]);
142 }
143
144 timer.Print("");
145
146 return res;
147 }

154 private static Vec3b ConvertColor(Vector3D c)
155 {
156 double r = Math.Min(254, Math.Max(0, c.X));
157 double g = Math.Min(254, Math.Max(0, c.Y));
158 double b = Math.Min(254, Math.Max(0, c.Z));
159 return new(Convert.ToByte(b), Convert.ToByte(g), Convert.ToByte(r));
160 }

177 private static void Kernel(Index1D index,
178 ArrayView<Ray> raysBuffer,
179 ArrayView<Triangle> trianglesBuffer,
180 ArrayView<GpuSubMesh> subMeshesBuffer,
181 ArrayView<LightSource> lightSourcesBuffer,
182 ArrayView<Material> materialsBuffer,
183 ArrayView<Vec3b> texturesDataBuffer,
184 ArrayView<GpuTextureMetadata> texturesMetadataBuffer,
185 ArrayView<GpuPhotonTreeNode> photonsTreeBuffer,
186 ArrayView<GpuPhotonNeighbor> photonsNearestNeighborsBuffer,
187 ArrayView<byte> photonsSearchStateBuffer,
188 ArrayView<Vector3D> pixelsBuffer)
189 {
190 int photonsTreeDepth = (int)(photonsSearchStateBuffer.Length / pixelsBuffer.Length);
191 int pixelIndex = index.X;
192 int firstRayIndex = pixelIndex * GlobalVariables.maxRaysPerPixel;
193 int nextRayIndex = firstRayIndex + 1;
194
195 pixelsBuffer[index.X] = new(0, 0, 0);
196 for (int rayIndex = firstRayIndex;
197 rayIndex < firstRayIndex + GlobalVariables.maxRaysPerPixel; rayIndex++)
198 {
199 Ray ray = raysBuffer[rayIndex];
200 Vector3D color = new(0, 0, 0);
201 if (CollisionFinder.GetFirstCollision(ray,
202 trianglesBuffer,
203 subMeshesBuffer,
204 out Triangle collisionTriangle,
205 out double collisionAbciss,
206 out double u,
207 out double v))
208 {
209 Point3D collisionPoint = ray.origin + collisionAbciss * ray.direction;
210 Vector3D collisionNormal = collisionTriangle.GetNormal(u, v);
211 Material collisionMaterial = materialsBuffer[collisionTriangle.materialIndex];
212 Vector3D collisionColor = GetTriangleColor(
213 collisionTriangle, u, v,
214 materialsBuffer,
215 texturesDataBuffer,
216 texturesMetadataBuffer);
217
218 color = ProcessRayCollision(pixelIndex,
219 rayIndex,
220 ref nextRayIndex,
221 photonsTreeDepth,
222 collisionMaterial,
223 collisionPoint,
224 collisionNormal,
225 collisionColor,
226 raysBuffer,
227 trianglesBuffer,
228 subMeshesBuffer,
229 lightSourcesBuffer,
230 photonsTreeBuffer,
231 photonsNearestNeighborsBuffer,
232 photonsSearchStateBuffer);
233 }
234
235 pixelsBuffer[index.X] += color;
236
237 if (rayIndex + 1 == nextRayIndex)
238 {
239 // No more rays to cast
240 break;
241 }
242 }
243 }

264 private static Vector3D ProcessRayCollision(int pixelIndex, int rayIndex, ref int nextRayIndex, int photonsTreeDepth,
265 Material collisionMaterial,
266 Point3D collisionPoint,
267 Vector3D collisionNormal,
268 Vector3D collisionColor,
269 ArrayView<Ray> raysBuffer,
270 ArrayView<Triangle> trianglesBuffer,
271 ArrayView<GpuSubMesh> subMeshesBuffer,
272 ArrayView<LightSource> lightSourcesBuffer,
273 ArrayView<GpuPhotonTreeNode> photonsTreeBuffer,
274 ArrayView<GpuPhotonNeighbor> photonsNearestNeighborsBuffer,
275 ArrayView<byte> photonsSearchStateBuffer)
276 {
277 Vector3D res = new(0, 0, 0);
278
279 Ray ray = raysBuffer[rayIndex];
280
281 // Phong + Shadows
282 for(int lightSourceIndex = 0; lightSourceIndex < lightSourcesBuffer.Length; lightSourceIndex ++)
283 {
284 LightSource lightSource = lightSourcesBuffer[lightSourceIndex];
285 PhongModel.Compute(
286 ray,
287 collisionMaterial,
288 collisionPoint,
289 collisionNormal,
290 collisionColor,
291 lightSource,
292 out Vector3D ambiantPart,
293 out Vector3D phongPart);
294
295 double shadowRatio = Shadow.Compute(
296 collisionPoint,
297 lightSource,
298 trianglesBuffer,
299 subMeshesBuffer);
300
301 res += (1 - shadowRatio) * phongPart + ambiantPart;
302 }
303
304 // Caustics
305 if(photonsTreeBuffer.Length > 0)
306 {
307 int nNeighborPhotons = NearestNeighborSearch.Run(
308 pixelIndex,
309 collisionPoint,
310 photonsTreeDepth,
311 photonsTreeBuffer,
312 photonsNearestNeighborsBuffer,
313 photonsSearchStateBuffer);
314
315 res += Caustic.Compute(pixelIndex * GlobalVariables.nNearestNeighbors,
316 nNeighborPhotons,
317 photonsNearestNeighborsBuffer);
318 }
319
320 // Reflection
321 if (Reflection.Compute(collisionMaterial, ray, collisionPoint, collisionNormal,
322 out Ray reflectedRay, out double reflectionCoeff))
323 {
324 if (ray.weight * reflectionCoeff > GlobalVariables.minReflectionRefractionCoeff)
325 {
326 res *= (1 - reflectionCoeff);
327 reflectedRay.weight = ray.weight * reflectionCoeff;
328 RecastRay(reflectedRay, ref nextRayIndex, raysBuffer);
329 }
330 }
331
332 // Refraction
333 if (Refraction.Compute(collisionMaterial, ray, collisionPoint, collisionNormal,
334 out Ray refractedRay, out double refractionCoeff))
335 {
336 if (ray.weight * refractionCoeff > GlobalVariables.minReflectionRefractionCoeff)
337 {
338 res *= (1 - refractionCoeff);
339 refractedRay.weight = ray.weight * refractionCoeff;
340 RecastRay(refractedRay, ref nextRayIndex, raysBuffer);
341 }
342 }
343
344 return ray.weight * res;
345 }

354 private static bool RecastRay(in Ray ray, ref int nextRayIndex, ArrayView<Ray> raysBuffer)
355 {
356 if ((nextRayIndex + 1) / GlobalVariables.maxRaysPerPixel ==
357 nextRayIndex / GlobalVariables.maxRaysPerPixel)
358 {
359 raysBuffer[nextRayIndex] = ray;
360 nextRayIndex++;
361 return true;
362 }
363 return false;
364 }

376 private static Vector3D GetTriangleColor(
377 Triangle triangle,
378 double u,
379 double v,
380 ArrayView<Material> materialsBuffer,
381 ArrayView<Vec3b> texturesDataBuffer,
382 ArrayView<GpuTextureMetadata> texturesMetadataBuffer)
383 {
384 if (triangle.textureIndex < 0)
385 {
386 return materialsBuffer[triangle.materialIndex].color;
387 }
388
389 double uVecX = u * triangle.B.texture.X + 0.5 * (1 - (u + v)) * triangle.A.texture.X;
390 double uVecY = u * triangle.B.texture.Y + 0.5 * (1 - (u + v)) * triangle.A.texture.Y;
391 double vVecX = v * triangle.C.texture.X + 0.5 * (1 - (u + v)) * triangle.A.texture.X;
392 double vVecY = v * triangle.C.texture.Y + 0.5 * (1 - (u + v)) * triangle.A.texture.Y;
393 Point2d textureCoordinates = new(uVecX + vVecX, uVecY + vVecY);
394
395 GpuTextureMetadata metadata = texturesMetadataBuffer[triangle.textureIndex];
396 int col = (int)(textureCoordinates.X * (metadata.width - 1));
397 int row = (int)((metadata.height - 1) - textureCoordinates.Y * (metadata.height - 1));
398 Vec3b color = texturesDataBuffer[metadata.dataIndex + row * metadata.width + col];
399 return new(color.Item2, color.Item1, color.Item0);
400 }
