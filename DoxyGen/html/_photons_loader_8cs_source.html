64 private static List<Photon> CreatePhotonEmissionPattern()
65 {
66 List<Photon> res = new(GlobalVariables.photonsSampling);
67 Photon p = new();
68
69 double dTheta = 2 * Math.PI / (GlobalVariables.photonsSampling);
70 double dPhi = Math.PI / (GlobalVariables.photonsSampling);
71
72 for (int i = 0; i < GlobalVariables.photonsSampling; i++)
73 {
74 double theta = i * dTheta;
75 double sinTheta = Math.Sin(theta);
76 double cosTheta = Math.Cos(theta);
77 for (int j = 0; j < GlobalVariables.photonsSampling; j++)
78 {
79 double phi = j * dPhi;
80 double sinPhi = Math.Sin(phi);
81 double cosPhi = Math.Cos(phi);
82 p.ray.direction.X = sinPhi * cosTheta;
83 p.ray.direction.Y = sinPhi * sinTheta;
84 p.ray.direction.Z = cosPhi;
85 res.Add(p);
86 }
87 }
88 return res;

128 private static int ConstructPhotonsTree(
129 PhotonTree tree,
130 List<GpuPhotonTreeNode> photonsTreeBuffer)
131 {
132 if(tree.count == 0) { return 0; }
133
134 int index = photonsTreeBuffer.Count;
135 photonsTreeBuffer.Add(new());
136
137 GpuPhotonTreeNode node;
138 node.data = tree.data;
139 node.axis = tree.axis;
140 node.parentIndex = -1;
141 node.leftIndex = -1;
142 node.rightIndex = -1;
143 if (tree.left != null)
144 {
145 node.leftIndex = ConstructPhotonsTree(tree.left, photonsTreeBuffer);
146 GpuPhotonTreeNode leftNode = photonsTreeBuffer[node.leftIndex];
147 leftNode.parentIndex = index;
148 photonsTreeBuffer[node.leftIndex] = leftNode;
149 }
150 if (tree.right != null)
151 {
152 node.rightIndex = ConstructPhotonsTree(tree.right, photonsTreeBuffer);
153 GpuPhotonTreeNode rightNode = photonsTreeBuffer[node.rightIndex];
154 rightNode.parentIndex = index;
155 photonsTreeBuffer[node.rightIndex] = rightNode;
156 }
157 photonsTreeBuffer[index] = node;
158
159 return index;
