113 private static bool CastPhoton(int photonIndex,
114 ArrayView<Triangle> trianglesBuffer,
115 ArrayView<GpuSubMesh> subMeshesBuffer,
116 ArrayView<Material> materialsBuffer,
117 ArrayView<Photon> photonsBuffer,
118 ArrayView<XorShift128Plus> randomBuffer)
119 {
120
121 // Find collision info
122 Photon photon = photonsBuffer[photonIndex];
123 if(!CollisionFinder.GetFirstCollision(photon.ray,
124 trianglesBuffer,
125 subMeshesBuffer,
126 out Triangle collisionTriangle,
127 out double collisionAbciss,
128 out double u, out double v))
129 {
130 // Photon does not hit anything, discard it
131 photon.isCaustic = 0;
132 photonsBuffer[photonIndex] = photon;
133 return true;
134 }
135 Point3D collisionPoint = photon.ray.origin + collisionAbciss * photon.ray.direction;
136 Vector3D collisionNormal = collisionTriangle.GetNormal(u, v);
137 Material collisionMaterial = materialsBuffer[collisionTriangle.materialIndex];
138
139 // compute reflection probability
140 double reflectionProbability = 0;
141 if(Reflection.Compute(collisionMaterial, photon.ray, collisionPoint, collisionNormal,
142 out Ray reflectedRay, out double reflectionCoeff))
143 {
144 reflectionProbability = reflectionCoeff;
145 }
146
147 // compute refraction probability
148 double refractionProbability = 0;
149 if (Refraction.Compute(collisionMaterial, photon.ray, collisionPoint, collisionNormal,
150 out Ray refractedRay, out double refractionCoeff))
151 {
152 refractionProbability = refractionCoeff;
153 }
154
155 // Throw a dice
156 XorShift128Plus rng = randomBuffer[photonIndex];
157 double rand = rng.NextDouble();
158 randomBuffer[photonIndex] = rng;
159 if (rand < reflectionProbability)
160 {
161 // Reflection
162 photon.ray = reflectedRay;
163 photon.isCaustic = 1;
164 photon.color *= reflectionProbability;
165 photonsBuffer[photonIndex] = photon;
166 return false;
167 }
168 if (rand < reflectionProbability + refractionProbability)
169 {
170 // Refraction
171 photon.ray = refractedRay;
172 photon.isCaustic = 1;
173 photon.color *= refractionProbability;
174 photonsBuffer[photonIndex] = photon;
175 return false;
176 }
177
178 // If the photon has not been refracted or reflected,
179 // update its position and color and stop propagating it
180 photon.color.X *= collisionMaterial.color.X / 255;
181 photon.color.Y *= collisionMaterial.color.Y / 255;
182 photon.color.Z *= collisionMaterial.color.Z / 255;
183 photon.ray.origin = collisionPoint;
184 photonsBuffer[photonIndex] = photon;
185 return true;
186 }
187 }
